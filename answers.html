inter

String Manipulation
Assignment Instructions Write a program that takes two string arguments and displays all of the characters that appear in both strings.
• Display order must be based on the first string.
• No duplicates.
• If args != 2, display nothing.
Correct Solution 
package main

import (
    "os"
    "github.com/01-edu/z01"
)

// Helper to avoid writing nested loops repeatedly
func Contains(str string, char rune) bool {
    for _, c := range str {
        if c == char {
            return true
        }
    }
    return false
}

func main() {
    if len(os.Args) != 3 {
        return
    }

    s1 := os.Args[1]
    s2 := os.Args[2]
    printed := ""

    for _, char := range s1 {
        // If char is in s2 AND we haven't printed it yet
        if Contains(s2, char) && !Contains(printed, char) {
            z01.PrintRune(char)
            printed += string(char)
        }
    }
    z01.PrintRune('\n')
}

union

String Manipulation
Assignment Instructions Write a program that takes two strings and displays characters that appear in either one.
• No duplicates.
• Order of first occurrence (start with s1, then s2).
• If args != 2, display newline.
Correct Solution 
package main

import (
    "os"
    "github.com/01-edu/z01"
)

func main() {
    if len(os.Args) != 3 {
        z01.PrintRune('\n')
        return
    }

    // Concatenate s1 and s2 to handle them as one sequence
    combined := os.Args[1] + os.Args[2]
    seen := make(map[rune]bool)

    for _, char := range combined {
        if !seen[char] {
            z01.PrintRune(char)
            seen[char] = true
        }
    }
    z01.PrintRune('\n')
}

hiddenp

Pointers / Logic
Assignment Instructions Check if the first string is hidden in the second one (subsequence).
• If found, print 1 followed by newline.
• If not, print 0 followed by newline.
Correct Solution 
package main

import (
    "os"
    "github.com/01-edu/z01"
)

func main() {
    if len(os.Args) != 3 {
        return
    }

    s1 := os.Args[1]
    s2 := os.Args[2]

    // Empty string is technically hidden in everything
    if s1 == "" {
        z01.PrintRune('1')
        z01.PrintRune('\n')
        return
    }

    i := 0
    for _, r := range s2 {
        if i < len(s1) && r == rune(s1[i]) {
            i++
        }
    }

    if i == len(s1) {
        z01.PrintRune('1')
    } else {
        z01.PrintRune('0')
    }
    z01.PrintRune('\n')
}

wdmatch

Pointers / Logic
Assignment Instructions Check if the first string is hidden in the second one.
• If possible, display the string itself followed by a newline.
• If not, display nothing.
Correct Solution 
package main

import (
    "os"
    "github.com/01-edu/z01"
)

func main() {
    if len(os.Args) != 3 {
        z01.PrintRune('\n')
        return
    }

    s1, s2 := os.Args[1], os.Args[2]
    runes1 := []rune(s1)
    count := 0

    for _, r := range s2 {
        // Critical Fix: Check count < length before accessing array
        if count < len(runes1) {
            if r == runes1[count] {
                count++
            }
        }
    }

    if count == len(runes1) {
        for _, r := range s1 {
            z01.PrintRune(r)
        }
        z01.PrintRune('\n')
    }
}

rstrcap

Formatting / Logic
Assignment Instructions Capitalize the last letter of each word.
• Lowercase the rest.
• "Word" = alphanumeric characters.
Correct Solution 
package main

import (
    "os"
    "github.com/01-edu/z01"
)

func isAlpha(r rune) bool {
    return (r >= 'a' && r <= 'z') || (r >= 'A' && r <= 'Z')
}

func toUpper(r rune) rune {
    if r >= 'a' && r <= 'z' { return r - 32 }
    return r
}

func toLower(r rune) rune {
    if r >= 'A' && r <= 'Z' { return r + 32 }
    return r
}

func main() {
    if len(os.Args) < 2 { return }

    for i, arg := range os.Args[1:] {
        runes := []rune(arg)
        for j, char := range runes {
            if isAlpha(char) {
                // Look Ahead: If next char is NOT alpha, this is the end of word
                if j == len(runes)-1 || !isAlpha(runes[j+1]) {
                    z01.PrintRune(toUpper(char))
                } else {
                    z01.PrintRune(toLower(char))
                }
            } else {
                z01.PrintRune(char)
            }
        }
        if i < len(os.Args)-2 {
            z01.PrintRune(' ')
        }
    }
    z01.PrintRune('\n')
}

saveandmiss

Loops
Assignment Instructions Write a function SaveAndMiss(arg string, num int) string.
• Save num characters.
• Skip num characters.
• Repeat.
Correct Solution 
package piscine

func SaveAndMiss(arg string, num int) string {
    if num <= 0 { return arg }
    var result string

    // Increment i by num * 2 to jump over the "Save" AND the "Miss" block
    for i := 0; i < len(arg); i += num * 2 {
        for j := i; j < i+num; j++ {
            if j < len(arg) {
                result += string(arg[j])
            }
        }
    }
    return result
}

fifthandskip

Loops / String
Assignment Instructions Write a function FifthAndSkip(str string) string.
• Remove spaces from input.
• If length < 5, return "Invalid Input".
• Take 5 chars, skip the 6th, add a space between groups.
Correct Solution 
package piscine

func FifthAndSkip(str string) string {
    // 1. Clean the string (Remove spaces)
    cleaned := ""
    if str == "" { return "\n" }
    for _, char := range str {
        if char != ' ' {
            cleaned += string(char)
        }
    }

    if len(cleaned) < 5 { return "Invalid Input\n" }

    result := ""
    count := 0

    for i := 0; i < len(cleaned); i++ {
        result += string(cleaned[i])
        count++

        if count == 5 {
            if i+1 < len(cleaned) {
                result += " "
                i++ // Skip the 6th character manually
            }
            count = 0
        }
    }

    return result + "\n"
}

revconcatalternate

Slices
Assignment Instructions Write a function RevConcatAlternate(slice1, slice2 []int) []int.
• Reverse the values of both slices.
• Alternate between them (starting with the longer slice).
• Append remaining values.
Correct Solution 
package piscine

func RevConcatAlternate(slice1, slice2 []int) []int {
    var result []int
    i := len(slice1) - 1
    j := len(slice2) - 1

    for i >= 0 || j >= 0 {
        // If s1 is longer (or equal), it goes first
        if len(slice1) >= len(slice2) {
            if i >= 0 { result = append(result, slice1[i]); i-- }
            if j >= 0 { result = append(result, slice2[j]); j-- }
        } else {
            // If s2 is longer
            if j >= 0 { result = append(result, slice2[j]); j-- }
            if i >= 0 { result = append(result, slice1[i]); i-- }
        }
    }
    return result
}

slice

Variadic / Logic
Assignment Instructions Write a function that mimics Python slicing.
• Takes slice and variadic ints (start, end).
• Handle negative indices (count from end).
• Clamp to valid bounds.
Correct Solution 
package piscine

func Slice(a []string, nbrs ...int) []string {
    start, end := 0, len(a)

    // Handle optional arguments
    if len(nbrs) > 0 { start = nbrs[0] }
    if len(nbrs) > 1 { end = nbrs[1] }

    // Handle negatives
    if start < 0 { start = len(a) + start }
    if end < 0 { end = len(a) + end }

    // Clamp
    if start < 0 { start = 0 }
    if end > len(a) { end = len(a) }
    
    // Invalid range
    if start > end { return nil }

    return a[start:end]
}
